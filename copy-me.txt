! Adjust grid display to avoid gaps  
www.youtube.com##ytd-rich-grid-row:style(display: flex !important; flex-wrap: wrap !important)  
www.youtube.com##ytd-rich-grid-renderer:style(--ytd-rich-grid-items-per-row: 4 !important)  

! Hide videos with Shorts indicator  
www.youtube.com##ytd-grid-video-renderer:has([overlay-style="SHORTS"]):style(display: none !important)  
www.youtube.com##ytd-rich-item-renderer:has([overlay-style="SHORTS"]):style(display: none !important)  
www.youtube.com##ytd-video-renderer:has([overlay-style="SHORTS"]):style(display: none !important)  

! Hide Shorts button in sidebar  
www.youtube.com##ytd-guide-entry-renderer:has(yt-formatted-string:has-text(/^Shorts$/i)):style(display: none !important)  
www.youtube.com##ytd-mini-guide-entry-renderer:has(.title:has-text(/^Shorts$/i)):style(display: none !important)  

! Hide Shorts section on the homepage  
www.youtube.com##ytd-rich-section-renderer:has(#title:has-text(/(^| )Shorts( |$)/i)):style(display: none !important)  
www.youtube.com##ytd-reel-shelf-renderer:has(.ytd-reel-shelf-renderer:has-text(/(^| )Shorts( |$)/i)):style(display: none !important)  

! Hide Shorts tab on channel pages  
www.youtube.com##tp-yt-paper-tab:has(.tp-yt-paper-tab:has-text(Shorts)):style(display: none !important)  
www.youtube.com##yt-tab-shape:has-text(/^Shorts$/):style(display: none !important)  

! Support for YouTube UI 2025  
www.youtube.com##ytd-rich-grid-row:has([class*="shorts"]):style(display: none !important)  
www.youtube.com##div[class*="shorts-container"]:style(display: none !important; height: 0 !important; overflow: hidden !important)  

! Mobile version  
m.youtube.com##ytm-pivot-bar-item-renderer:has(.pivot-shorts):style(display: none !important)  
m.youtube.com##ytm-video-with-context-renderer:has([data-style="SHORTS"]):style(display: none !important)  
m.youtube.com##ytm-rich-section-renderer:has(.reel-shelf-title-wrapper .yt-core-attributed-string:has-text(/(^| )Shorts( |$)/i)):style(display: none !important)  
m.youtube.com##yt-tab-shape:has-text(/^Shorts$/):style(display: none !important)
