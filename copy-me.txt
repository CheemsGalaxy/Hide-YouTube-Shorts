!!! DESKTOP !!!
! Fix for black spaces when hiding shorts - adjust grid layout
www.youtube.com##ytd-rich-grid-row:style(display: flex !important; flex-wrap: wrap !important)
www.youtube.com##ytd-rich-grid-renderer:style(--ytd-rich-grid-items-per-row: 4 !important)
www.youtube.com##ytd-rich-grid-renderer:style(--ytd-rich-grid-posts-per-row: 4 !important)
www.youtube.com##ytd-rich-grid-renderer:style(--ytd-rich-grid-shorts-per-row: 4 !important)

! Fix empty spaces in grid with proper collapsing
www.youtube.com##ytd-rich-grid-row,#contents.ytd-rich-grid-row:style(display: contents !important)

! Hide all videos containing the phrase "#shorts"
www.youtube.com##ytd-grid-video-renderer:has(#video-title:has-text(/(^| )#Shorts?( |$)/i)):remove()
www.youtube.com##ytd-rich-item-renderer:has(#video-title:has-text(/(^| )#Shorts?( |$)/i)):remove()

! Hide all videos with the shorts indicator on the thumbnail
www.youtube.com##ytd-grid-video-renderer:has([overlay-style="SHORTS"]):remove()
www.youtube.com##ytd-rich-item-renderer:has([overlay-style="SHORTS"]):remove()
www.youtube.com##ytd-video-renderer:has([overlay-style="SHORTS"]):remove()
www.youtube.com##ytd-item-section-renderer.ytd-section-list-renderer[page-subtype="subscriptions"]:has(ytd-video-renderer:has([overlay-style="SHORTS"])):remove()

! Hide shorts button in sidebar
www.youtube.com##ytd-guide-entry-renderer:has(yt-formatted-string:has-text(/^Shorts$/i)):remove()

! Tablet resolution
www.youtube.com##ytd-mini-guide-entry-renderer:has(.title:has-text(/^Shorts$/i)):remove()

! Hide shorts section on homepage with removal instead of hiding
www.youtube.com##ytd-rich-section-renderer:has(#title:has-text(/(^| )Shorts( |$)/i)):remove()
www.youtube.com##ytd-reel-shelf-renderer:has(.ytd-reel-shelf-renderer:has-text(/(^| )Shorts( |$)/i)):remove()

! Hide shorts tab on channel pages (old + new styles)
www.youtube.com##tp-yt-paper-tab:has(.tp-yt-paper-tab:has-text(Shorts)):remove()
www.youtube.com##yt-tab-shape:has-text(/^Shorts$/):remove()

! Hide short remixes in video descriptions
www.youtube.com##ytd-reel-shelf-renderer.ytd-structured-description-content-renderer:has-text(/^Shorts remixing this video$/i):remove()

! Hide shorts category on homepage and search pages
www.youtube.com##yt-chip-cloud-chip-renderer:has(yt-formatted-string:has-text(/^Shorts$/i)):remove()

! Fix for shorts sections (remove completely instead of hiding)
www.youtube.com##.ytd-rich-grid-group-contents:remove()
www.youtube.com##.ytd-rich-grid-group-host:remove()
www.youtube.com##.ytdRichGridGroupHostFirstColumn.ytdRichGridHost > .ytdRichGridGroupContents:remove()
www.youtube.com##.ytdRichGridGroupContents:remove()
www.youtube.com##.ytdRichGridGroupHost:remove()

! Additional fixes for 2025 YouTube UI with removal
www.youtube.com##ytd-shelf-renderer:has(.ytd-shelf-renderer:has-text(/Shorts/)):remove()
www.youtube.com##ytd-rich-grid-renderer:has(.ytd-rich-section-renderer:has-text(/Shorts/)):remove()

! Force grid recalculation after shorts removal
www.youtube.com##ytd-browse:style(display: flex !important; flex-direction: column !important)
www.youtube.com##ytd-two-column-browse-results-renderer:style(display: flex !important; flex-direction: column !important)

! 2025 extra selectors to fully remove Shorts containers
www.youtube.com##ytd-rich-grid-row:has([class*="shorts"]):remove()
www.youtube.com##div[class*="shorts-container"]:remove()
www.youtube.com##div[id*="shorts-container"]:remove()
